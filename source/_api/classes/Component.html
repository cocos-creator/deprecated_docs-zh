cancelInvoke
(
  • methodNameOrInvokeId
)

定义于 utils/api/engine/component.js:373

Cancels previous invoke calls with methodName or InvokeID on this component. When using methodName, all calls with the same methodName will be canceled. InvokeID is the identifier of the invoke action you want to cancel, as returned by invoke().

参数列表:

  • methodNameOrInvokeId String | number

例子:

  // ...........
  constructor: function() {
    // declare a variable to store invoke id across different methods
    this.invoke1 = null;
  },

  // start two invokes, one with method name, one with function
  startInvokes: function() {
    // use invoke to delay a function call
    this.invoke1 = this.invoke(function() {
      Fire.log("This message will show in 2 seconds");
    }, 2);
    // invoke by a method name
    this.invoke("invoke2", 1);
  },

  invoke2: function () {
    Fire.log("This is called with a method name. Will show up in 1 second.");
  },

  // we will cancel the two invokes here
  cancel: function () {
    // cancel invoke1 by id
    this.cancelInvoke(this.invoke1);
    // cancel invoke2 by name
    this.cancelInvoke("invoke2");
  }
});

定义于 utils/api/engine/component.js:392

Cancels previous repeat calls with methodName or RepeatID on this component. When using methodName, all calls with the same methodName will be canceled. RepeatID is the identifier of the repeat action you want to cancel, as returned by repeat().

参数列表:

  • methodNameOrRepeatId String | number

例子:

  // ...........
  constructor: function() {
    // declare a variable to store repeat id across different methods
    this.repeat1 = null;
  },

  // start two repeats, one with method name, one with function
  startRepeats: function() {
    // use repeat to repeatedly call a function with a delay in between
    this.repeat1 = this.repeat(function() {
      Fire.log("This message will show in EVERY 2 seconds");
    }, 2);
    // repeat by a method name
    this.repeat("repeat2", 1);
  },

  repeat2: function () {
    Fire.log("This is called with a method name. Will show in EVERY second.");
  },

  // we will cancel the two repeats here
  cancel: function () {
    // cancel repeat1 by id
    this.cancelRepeat(this.repeat1);
    // cancel repeat2 by name
    this.cancelRepeat("repeat2");
  }
});

invoke
(
  • functionOrMethodName
  • [delay=0]
)
Number

定义于 utils/api/engine/component.js:347

Invokes the method on this component after a specified delay. The method will be invoked even if this component is disabled, but will not invoked if this component is destroyed.

参数列表:

  • functionOrMethodName Function | string
  • delay Number optional ，默认值：0

    The number of seconds that the function call should be delayed by. If omitted, it defaults to 0. The actual delay may be longer.

返回值:

类型： Number

Will returns a new InvokeID if the functionOrMethodName is type function. InvokeID is the numerical ID of the invoke, which can be used later with cancelInvoke().

例子:

  // ...........

  //  use this for initialization
  start: function () {
      // use invoke to delay a function call
      this.invoke(function() {
          Fire.log("This message will show in 2 seconds");
          // you can also invoke a instance method name,
          // the second parameter is the delay before function call
          var delay = 1; //in seconds
          this.invoke('hello', delay);
      }, 2);
  },

  // define instance method for invoke
  hello: function () {
      Fire.log("Hello!");
  }
});

参数列表:

  • functionOrMethodName Function | string
  • delay Number optional ，默认值：0

返回值:

  // ...........

  start: function () {
      // Repeatedly to call a named or anonymous function with a delay between each calls

          Fire.log("This message will show in every 2 seconds");
      }, 2);
  }
